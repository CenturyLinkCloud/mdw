Here's what JAVA_OPTS looks like in my local catalina.bat:
set CFG_DIR=%CATALINA_HOME%/mdw/config
set CFG_OPTIONS=-Dmdw.config.location=%CFG_DIR%
set RUNTIME_ENV=dev
set MDW_OPTIONS=-DruntimeEnv=%RUNTIME_ENV%
set CT_OPTIONS=-Dcom.qwest.appsec.CTECOMFilterConfigFilePath=%CFG_DIR%/CTECOMFilter.config -Dcom.qwest.appsec.CTAPPFilterConfigFilePath=%CFG_DIR%/CTAPPFilter.config
set JMX_OPTIONS=-Dcom.sun.management.jmxremote.port=8501 -Dcom.sun.management.jmxremote.ssl=false -Dcom.sun.management.jmxremote.authenticate=false
set LOG_OPTIONS=-Djava.util.logging.config.file=%CFG_DIR%\tomcat.logging.properties
set VVM_OPTIONS=-Dcom.sun.management.jmxremote -Dvisualvm.id=7768877011497109
set DEBUG_OPTS=-Xdebug -Xrunjdwp:transport=dt_socket,server=y,suspend=n,address=8006
set JAVA_OPTS=%CFG_OPTIONS% %TM_OPTIONS% %CT_OPTIONS% %JMX_OPTIONS% %LOG_OPTIONS% %VVM_OPTIONS% %DEBUG_OPTS%
export JAVA_OPTS="-DruntimeEnv=dev -Dmdw.config.location=/foss/foss-ews/instances/mdwapp/current/Qwest/config -Dcom.qwest.appsec.CTECOMFilterConfigFilePath=/foss/foss-ews/instances/mdwapp/current/Qwest/config/CTECOMFilter.config -Dcom.qwest.appsec.CTAPPFilterConfigFilePath=/foss/foss-ews/instances/mdwapp/current/Qwest/config/CTAPPFilter.config -Xdebug -Xrunjdwp:transport=dt_socket,server=y,suspend=n,address=8006"
